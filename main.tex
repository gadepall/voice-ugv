\begin{abstract}
This paper implements a cloud-free voice-controlled Unmanned Ground
Vehicle (UGV), consisting of a simple speech recognition engine and microcontroller based mobility.
The novelty of this work lies in the fact that
all 
voice commands are processed 
on an Android
smartphone 
where  all speech recognition algorithms are implemented.
The key algorithms used are Mel-Frequency Cepstral Coefficient (MFCC) features
combined with Dynamic Time Warping (DTW).
User adaptation is supported through up to ten recorded templates
per command, achieving 90--95\% accuracy across 10 speakers and demonstrating
language-agnostic operation in English, Hindi, Gujarati, and Tamil.

\section{Introduction} \label{sec:intro}

Speech-based control removes the need for physical controllers
and lowers the barrier for non-expert operation of ground robots.
Most prior implementations use recognition to cloud automatic speech recognition (ASR)
services~\cite{li2017speech}, which introduces round-trip
latency and ties the system to network availability. Running
a full speech model on the robot microcontroller directly is
equally impractical: a device such as the ESP32 carries
insufficient flash and RAM for any modern recogniser.

This paper resolves the compute--connectivity dilemma by
offloading all recognition to the Android smartphone that
the operator already carries. Short utterances are converted
to MFCC feature matrices~\cite{davis1980mfcc} and matched
against stored templates using DTW~\cite{sakoe1978dtw}.
Both steps run entirely on the phone, so no audio ever leaves
the device. Neural keyword-spotting alternatives exist for
microcontrollers~\cite{zhang2017hello, choi2019iot} but demand
labelled training corpora and quantisation pipelines that a
DTW system avoids entirely. A 7-byte bluetooth low energy (BLE) command string is
transmitted to the ESP32 only after a match is confirmed,
reducing the wireless payload by over 99\% compared to
streaming raw audio~\cite{mikhaylov2013ble}.
\subsection{Key Contributons}
The specific contributions of this work are
\begin{enumerate}
  \item A smartphone-resident MFCC+DTW pipeline that operates
    without any cloud or internet dependency.
  \item A Best-of-$N$ template strategy allowing each user to
    enrol up to ten personal utterances per command for
    speaker adaptation.
  \item A latency-optimised BLE scheme that cuts end-to-end
    response time from 2--4\,s to approximately 1.6\,s by
    transmitting only the recognised command label.
\end{enumerate}

\section{Recognition Algorithm} \label{sec:recognition}

This section presents the complete end-to-end voice recognition pipeline as
formal algorithms. Algorithm~\ref{alg:mfcc} details MFCC extraction and
Algorithm~\ref{alg:dtw} details DTW recognition with Best-of-N
template matching.
% Best-of-N template matching.
\vspace{5pt}
\begin{algorithm}[H]
\caption{MFCC Feature Extraction}
\label{alg:mfcc}
\begin{algorithmic}[1]
\REQUIRE Raw audio samples $x[n]$, $n=0,\ldots,11999$ (1.5\,s at 8\,kHz)
\ENSURE Feature matrix $\hat{\mathbf{C}} \in \mathbb{R}^{92 \times 26}$

\STATE \textbf{Pre-emphasis:} $y[n] \leftarrow x[n] - 0.97\,x[n-1]$; normalize by $32768$
\FOR{each frame $f = 0$ \TO $91$}
  \STATE Extract samples $y[fH : fH+N]$ with $N=256$, $H=128$
  \STATE Apply Hamming window: $s[n] \leftarrow y[n]\cdot w[n]$
  \STATE Compute 256-point FFT; form power spectrum $P[k] = |X[k]|^2/N$
  \STATE Apply 26-band Mel filterbank; compute log energies $E[m]$
  \STATE Apply DCT-II to $E[m]$; retain first 13 coefficients $\mathbf{c}[f]$
\ENDFOR
\FOR{each frame $f = 0$ \TO $91$}
  \STATE Compute delta: $\Delta c[f][k] \leftarrow (c[f{+}1][k] - c[f{-}1][k])/2$
  \STATE Concatenate: $\mathbf{c}[f] \leftarrow [\mathbf{c}[f] \;\|\; \Delta\mathbf{c}[f]]$ (26-dim)
\ENDFOR
\STATE Compute per-coefficient mean $\mu_k$ and std $\sigma_k$ across all frames
\STATE CMVN: $\hat{c}[f][k] \leftarrow (c[f][k] - \mu_k)/(\sigma_k + 10^{-8})$
\RETURN $\hat{\mathbf{C}}$
\end{algorithmic}
\end{algorithm}
% \vspace{-16pt}
\begin{algorithm}[H]
\caption{Voice Command Recognition via DTW + Best-of-N}
\label{alg:dtw}
\begin{algorithmic}[1]
\REQUIRE Query MFCC matrix $\mathbf{Q} \in \mathbb{R}^{F_1 \times 26}$,
         template sets $\{\mathbf{T}_{c,i}\}$ for each command $c$,
         threshold $\tau = 12.0$
\ENSURE Recognized command $\hat{c}$, or \textsc{reject}

\FOR{each command $c \in \{$\textit{forward, back, left, right, stop}$\}$}
  \STATE $s_c \leftarrow \infty$
  \FOR{each stored template $\mathbf{T}_{c,i}$ of length $F_2$}
    \STATE Initialize $D_{\text{prev}}[j] \leftarrow \infty$,
           $L_{\text{prev}}[j] \leftarrow 0$ for $j=0,\ldots,F_2{-}1$
    \FOR{$i = 0$ \TO $F_1 - 1$}
      \FOR{$j = 0$ \TO $F_2 - 1$}
        \STATE $d \leftarrow \|\mathbf{Q}[i] - \mathbf{T}_{c,i}[j]\|_2$
        \STATE Let $a\!=\!D[i{-}1][j]$, $b\!=\!D[i][j{-}1]$,
               $c\!=\!D[i{-}1][j{-}1]$; $p^*\!\leftarrow\!\arg\min(a,b,c)$
        \STATE $D_{\text{curr}}[j] \leftarrow d + D_{p^*}$;\quad
               $L_{\text{curr}}[j] \leftarrow 1 + L_{p^*}$
      \ENDFOR
      \STATE $D_{\text{prev}} \leftarrow D_{\text{curr}}$;\quad
             $L_{\text{prev}} \leftarrow L_{\text{curr}}$
    \ENDFOR
    \STATE $\text{score} \leftarrow D_{\text{prev}}[F_2{-}1]\,/\,L_{\text{prev}}[F_2{-}1]$
    \STATE $s_c \leftarrow \min(s_c,\;\text{score})$ \hfill\COMMENT{Best-of-N}
  \ENDFOR
\ENDFOR
\STATE $\hat{c} \leftarrow \arg\min_c\; s_c$
\IF{$s_{\hat{c}} < \tau$}
  \RETURN $\hat{c}$
\ELSE
  \RETURN \textsc{reject}
\ENDIF
\end{algorithmic}
\end{algorithm}

\section{Software Implementation} \label{sec:software}

\subsection{ESP32 Firmware}

Three source files make up the firmware, built with PlatformIO
on the Arduino framework.

\textbf{ble\_receive.cpp} sets up a BLE GATT server with a WRITE
characteristic (\texttt{UUID:~...9002}) for incoming command strings
and a NOTIFY characteristic (\texttt{UUID:~...9003}) for
acknowledgements. Any string not matching the five known commands
is discarded. Advertising for new connection restarts automatically on disconnect.

\textbf{motor.cpp} toggles four GPIO pins (32, 33, 25, 26) to drive
the L293D. No PWM is used.

\textbf{main.cpp} implements the command dispatcher:

\begin{lstlisting}[language=C++, caption={ESP32 command dispatcher (main.cpp)}]
void loop() {
  char cmd[16] = {0};
  if (!ble_receive_command(cmd, sizeof(cmd)))
    { delay(5); return; }

  if (!strcmp(cmd,"forward")) { motor_forward(); }
  else if (!strcmp(cmd,"back")) { motor_back(); }
  else if (!strcmp(cmd,"left"))
    { motor_left(); delay(600); motor_stop(); }
  else if (!strcmp(cmd,"right"))
    { motor_right(); delay(600); motor_stop(); }
  else if (!strcmp(cmd,"stop")) { motor_stop(); }}
\end{lstlisting}

\textit{forward} and \textit{back} run until a \textit{stop} is
received. \textit{left} and \textit{right} run for 600\,ms then
auto-stop, giving approximately a 90° turn. The duration is
adjustable for different chassis.

\subsection{Android Application}
The app records 1.5\,s at 8\,kHz, runs MFCC extraction and DTW
matching on a background thread, and posts results to the UI thread.
A score below $\tau$ triggers a BLE write of the command string.
A debug screen displays per-command DTW scores live; a training
screen lets users record additional templates. A 200\,ms pause
before each \texttt{AudioRecord} initialisation prevents microphone
release errors on back-to-back recordings.

\subsection{Template Conversion Script}
\texttt{convert\_templates.py} is a one-time script that reads the
C header \texttt{templates.h}, strips the \texttt{f} suffix from
float literals, and writes one \texttt{.bin} file per command for
\texttt{TemplateStore} using \texttt{struct.pack}.

\subsection{Out-of-the-Box Deployment}
Flashing the firmware and installing the APK is all that is required.
The app scans for a BLE device named \texttt{ESP32\_UGV} and connects
automatically. Once paired, the user speaks a command and the vehicle
responds within approximately 1.6\,s.

\section{User Adaptation and Tuning} \label{sec:tuning}

\subsection{Recording Additional Templates}

The \texttt{TemplateStore} always holds template index~0, which ships
pre-built inside \texttt{assets} and was recorded by the original
developer. Slots 1--9 are filled on the target device through
\texttt{DebugActivity}'s record button; no PC or cable is needed after
the initial APK install. How many slots to fill depends on the use
case --- Table~\ref{tab:samples} gives concrete guidance.

\subsection{Threshold Tuning}

Picking a good $\tau$ is straightforward with the live score display
in the debug screen. Run several genuine utterances and note the
highest DTW score that still matches the correct command; then run
a few wrong commands and note the lowest score that correctly rejects
them. Set $\tau$ anywhere in the gap between those two numbers. In
practice, 8--10 personal templates recorded in a quiet room push the
correct-command scores down to the 3.0--7.5 range (Table~\ref{tab:results}),
so a threshold of 8.0--10.0 gives comfortable separation for a single
speaker.

\subsection{Turn Duration}

Lateral movement is timed rather than angle-sensed: \texttt{main.cpp}
holds the turn GPIO pattern for 600\,ms before issuing a stop, which
yields roughly 90° on the development chassis at full motor voltage.
Heavier or slower builds need a longer delay; faster or lighter ones
need a shorter one. The constant is a single integer literal in
\texttt{main.cpp} and requires no recompilation of any Android code.

\subsection{Multi-Speaker Evaluation}

Ten volunteers each recorded 10 templates per command through the
in-app training interface, then independently spoke each command
multiple times in a quiet indoor room. Across all speakers and all
five commands the system maintained \textbf{90--95\%} accuracy, as
broken down in Table~\ref{tab:results}.

\input{tables/table_results.tex}

DTW is fundamentally a speaker-dependent algorithm, so achieving this
accuracy across 10 different voices warrants explanation. Two
mechanisms are responsible. First, CMVN (Equation~\ref{eq:cmvn})
subtracts the per-coefficient mean and divides by the standard
deviation computed over the 92-frame window, stripping out
slow-varying spectral biases that differ between speakers. Second, the
Best-of-N rule (Equation~\ref{eq:bestofn}) picks the single closest
template out of up to 10 stored ones, so each user's personal
recordings effectively anchor the matcher to their own vocal
characteristics. 

\subsection{Cross-Language Evaluation}

A subset of participants issued commands in \textbf{Hindi, Gujarati,
and Tamil} rather than English. Accuracy stayed within the 90--95\%
band across all three languages. No changes to the pipeline were
needed: because \texttt{MFCCProcessor} works on raw acoustic energy
and \texttt{DTWMatcher} compares 26-dimensional frame vectors, the
system carries no phoneme tables, pronunciation dictionaries, or
language models that would tie it to a specific language. Any
consistently pronounced word maps to a repeatable MFCC trajectory and
can therefore serve as a command.

\subsection{Latency}

Button press to motor response measured at roughly 1.6\,s. The
previous ESP32-resident pipeline took 2--4\,s because it had to
receive all 24{,}000 bytes of 8\,kHz PCM over BLE before extraction
could begin. Shifting MFCC and DTW to the phone reduces the BLE
transfer to a 7-byte ASCII string, and that single change accounts for
most of the latency improvement.

\subsection{Limitations}

Each new speaker must record their own templates; there is currently
no speaker-independent enrolment path. In noisy settings the
\texttt{NoiseSuppressor} API offers some relief but does not fully
recover accuracy. On the ESP32 side, the 600\,ms fixed turn timer is
a heuristic --- it produces approximately 90° on the test chassis but
needs manual adjustment for any build with different motor speed or
chassis weight. Addressing all three points --- speaker independence,
noise robustness, and closed-loop turning --- is left to future work.

\section{Conclusions} \label{sec:conclusions}

Relocating MFCC extraction and DTW matching from the ESP32 to an
Android smartphone produces a fully on-device, cloud-free voice
control system for a UGV. The shift cuts end-to-end latency from
2--4\,s to 1.6\,s and recovers 80\,kB of microcontroller flash.
CMVN normalisation and the Best-of-$N$ template strategy together
yield 90--95\% accuracy across 10 speakers and four languages
— English, Hindi, Gujarati, and Tamil — despite DTW being
inherently speaker-dependent.

Two limitations remain. Each new speaker must record personal
templates, and accuracy drops in noisy environments even with
Android's \texttt{NoiseSuppressor} API active. Future work will
investigate compact convolutional or recurrent keyword-spotting
models that run on-device without per-user enrolment, alongside
data augmentation for noise robustness and a larger-scale
speaker evaluation.

\section{Mathematical Derivations}
\label{app:math}

This appendix collects all equations referenced in
Sections~\ref{sec:mfcc} and~\ref{sec:dtw}.

\subsection*{B.1\quad MFCC Feature Extraction}

\textbf{Pre-emphasis:}
\begin{equation}
  y[n] = x[n] - \alpha\, x[n-1], \quad \alpha = 0.97
  \label{eq:preemphasis}
\end{equation}

\textbf{Hamming window:}
\begin{equation}
  w[n] = 0.54 - 0.46\cos\!\left(\frac{2\pi n}{N-1}\right),
  \quad n = 0,\ldots,N-1
  \label{eq:hamming}
\end{equation}

\textbf{Power spectrum:}
\begin{equation}
  P[k] = \frac{|X[k]|^2}{N}, \quad k = 0,\ldots,\frac{N}{2}
  \label{eq:powspec}
\end{equation}

\textbf{Mel scale:}
\begin{equation}
  m = 2595\log_{10}\!\left(1 + \frac{f}{700}\right)
  \label{eq:meltohz}
\end{equation}

\textbf{Mel filterbank log energy:}
\begin{equation}
  E[m] = \log\!\left(\sum_{k} P[k]\,H_m[k]\right),
  \quad m = 0,\ldots,25
  \label{eq:melbank}
\end{equation}

\textbf{DCT-II coefficients:}
\begin{equation}
  c[k] = w_k \sum_{n=0}^{M-1} E[n]
         \cos\!\left(\frac{\pi k (2n+1)}{2M}\right),
  \quad k = 0,\ldots,12
  \label{eq:dct}
\end{equation}
\begin{equation}
  w_k = \begin{cases} \sqrt{1/M} & k=0 \\ \sqrt{2/M} & k>0 \end{cases}
  \label{eq:dctweight}
\end{equation}
where $M = 26$.

\textbf{Delta coefficients:}
\begin{equation}
  \Delta c[f][k] = \frac{c[f{+}1][k] - c[f{-}1][k]}{2}
  \label{eq:delta}
\end{equation}

\textbf{CMVN:}
\begin{align}
  \mu_k     &= \frac{1}{F}\sum_{f=1}^{F} c[f][k] \label{eq:mean}\\
  \sigma_k  &= \sqrt{\frac{1}{F}\sum_{f=1}^{F}
               \bigl(c[f][k]-\mu_k\bigr)^2} \label{eq:std}\\
  \hat{c}[f][k] &= \frac{c[f][k] - \mu_k}{\sigma_k + \varepsilon}
  \label{eq:cmvn}
\end{align}
where $\varepsilon = 10^{-8}$.

\subsection*{B.2\quad Dynamic Time Warping}

\textbf{Local distance:}
\begin{equation}
  d(i,j) = \left\|\mathbf{Q}[i] - \mathbf{T}[j]\right\|_2
  \label{eq:euclid}
\end{equation}

\textbf{Accumulated cost recurrence:}
\begin{equation}
  D[i][j] = d(i,j) + \min\bigl(D[i{-}1][j],\;
             D[i][j{-}1],\; D[i{-}1][j{-}1]\bigr)
  \label{eq:dtw}
\end{equation}

\textbf{Path-length tracking:}
\begin{equation}
  L[i][j] = 1 + L[\,p^*]
  \label{eq:pathlen}
\end{equation}
where $p^* = \arg\min\bigl(D[i{-}1][j],\; D[i][j{-}1],\; D[i{-}1][j{-}1]\bigr)$.

\textbf{Normalized DTW score:}
\begin{equation}
  \text{score} = \frac{D[F_1{-}1][F_2{-}1]}{L[F_1{-}1][F_2{-}1]}
  \label{eq:dtwnorm}
\end{equation}

\textbf{Best-of-N command score:}
\begin{equation}
  s(\text{cmd}) = \min_{i=0}^{N-1} \text{score}(\mathbf{Q},\,\mathbf{T}_i)
  \label{eq:bestofn}
\end{equation}

\textbf{Decision rule:}
\begin{equation}
  \hat{c} = \underset{\text{cmd}}{\arg\min}\; s(\text{cmd}),
  \quad \text{if } \min_{\text{cmd}} s(\text{cmd}) < \tau
  \label{eq:decision}
\end{equation}

\section{Software Installation}

building and installing the Android application.

\subsection*{A.1\quad Prerequisites}

The following tools must be installed on the host PC before proceeding:

\begin{itemize}
  \item \textbf{Python 3.8+} — required for template conversion and
    PlatformIO's build system. Available at \url{https://www.python.org}.
  \item \textbf{Git} — for cloning the repository.
    Available at \url{https://github.com/gadepall/voice-ugv.git}.
  \item \textbf{VS Code} (recommended) or any editor with PlatformIO support.
\end{itemize}

\subsection*{A.2\quad ESP32 Firmware} \label{app:esp32}

\begin{enumerate}
  \item \textbf{Install PlatformIO.}
    Install the PlatformIO IDE extension in VS Code, or install the
    PlatformIO Core CLI via:
\begin{lstlisting}[language=bash]
pip install platformio
\end{lstlisting}

  \item \textbf{Clone / extract the project.}
    Place the \texttt{ugv\_esp32/} folder on your PC.

%   \item \textbf{Generate MFCC template binaries.}
%     Run the conversion script once to produce the \texttt{.bin} files

% cd ugv_esp32

  \item \textbf{Connect the ESP32} to the PC via USB.

  \item \textbf{Build and flash} the firmware:
\begin{lstlisting}[language=bash]
pio run --target upload
\end{lstlisting}
    PlatformIO automatically installs the \texttt{ESP32 BLE Arduino}

    After flashing, the ESP32 advertises itself as \texttt{ESP32\_UGV}

  \item \textbf{Verify} by opening the serial monitor

    message \texttt{BLE started, advertising...}.

\subsection*{A.3\quad Android Application} \label{app:android}

\subsubsection*{Option A — Install the pre-built APK (recommended)}

  \item Copy \texttt{ugv\_android.apk} to the Android device.

\begin{enumerate}
  \item \textbf{Install Android Studio} (Hedgehog or later) from
    \url{https://developer.android.com/studio}.
    During setup, install the Android SDK (API level 33+) and
    the Android NDK if prompted.

  \item \textbf{Open the project.}
    In Android Studio, choose \textit{File $\to$ Open} and select the
    \texttt{ugv\_android/} folder. Wait for Gradle sync to complete.

  \item \textbf{Build the APK.}

\subsection*{A.4\quad First-Time Setup on the Phone}

\begin{enumerate}
  \item Enable Bluetooth on the Android device.
  \item Open the \textbf{UGV Controller} app.
    The \textit{SetupActivity} wizard launches automatically on first
    run, guiding you through BLE pairing.
  \item Tap \textit{Connect to ESP32}. The app scans for a BLE device
    named \texttt{ESP32\_UGV} and connects automatically.
  \item Follow the on-screen prompts to record 5 templates per command
    (10 recommended for best accuracy).
  \item Tap any command button to issue voice commands. The recognized
    command and DTW score are shown on the main screen; detailed
    per-command scores are available in the \textit{Debug} screen.
\end{enumerate}

\subsection*{A.5\quad Troubleshooting}

\begin{itemize}
  \item \textbf{AudioRecord failed:} A 200\,ms delay is already applied
    between recordings. If the error persists, close other apps
    using the microphone.
  \item \textbf{BLE not connecting:} Ensure the ESP32 is powered and
    advertising (\texttt{ESP32\_UGV} appears in a BLE scanner app).
    Force-stop the UGV app and reconnect.
  \item \textbf{Poor recognition accuracy:} Record additional templates
    in a quiet environment. Reduce the threshold $\tau$ in
    \texttt{DTWMatcher.java} if valid commands are being rejected.
  \item \textbf{Gradle build errors:} Ensure JDK 17 is selected in
    \textit{File $\to$ Project Structure $\to$ SDK Location}.
\end{itemize}
